%let wrds=wrds.wharton.upenn.edu 4016;
options comamid=TCP remote=WRDS;
signon user=_prompt_;

*FROM SCHOOL*;
libname rpo 'C:\Users\tz25\Downloads\RPOData';

* Reference macros for use later;
options errors=3 noovp; options nocenter ps=max ls=78; options mprint source nodate symbolgen macrogen;
options msglevel=i;
*WHEN AT SCHOOL;
%include "C:\Users\tz25\Downloads\Landon Seminar\Macro Code (2).sas";

rsubmit;

data comp;
	set compa.funda (keep = gvkey cusip tic fyear fyr sich cik datadate
		at revt oib aedi drc drlt ui urevub recub DATAFMT INDFMT POPSRC CONSOL CURCD);
	if fyear ge 2010;
	if DATAFMT	=	'STD';
	if INDFMT	=	'INDL';
	if POPSRC	=	'D';
	if CONSOL	=	'C';
	if CURCD	=	'USD';
run;

proc sort data=comp out=comp_gvkey_cik_fyear(keep=gvkey cusip tic cik fyear) nodupkey;
	by gvkey cik fyear;
run;

proc download data=comp_gvkey_cik_fyear;
run;

endrsubmit;

proc sort data=comp_gvkey_cik_fyear;
	by gvkey fyear;
run;

/* 把 cik 转成固定10位字符串，避免前导零丢失 */
data comp_gvkey_cik_fyear2;
	set comp_gvkey_cik_fyear;
	cik_str = put(input(cik, 10.), z10.);
run;

/* 导出成 CSV，路径写你本地电脑的路径 */
proc export data=comp_gvkey_cik_fyear2
	outfile="C:\Users\tz25\Downloads\RPOData\gvkey_cik_fyear.csv"
	dbms=csv
	replace;
run;
